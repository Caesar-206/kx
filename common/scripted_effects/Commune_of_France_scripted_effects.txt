FRA_CGT_effect = {
	if = {
		limit = {
			NOT = { has_country_flag = FRA_CGT_anarchosyn }
		}
		add_to_variable = {
			var = FRA_CGT_SYN_strength_var
			value = FRA_size_var
			tooltip = FRA_CGT_SYN_tt
		}
		add_to_variable = {
			var = FRA_CGT_ANAR_strength_var
			value = FRA_size_var
			tooltip = FRA_CGT_ANAR_tt
		}
		if = {
			limit = {
				NOT = { has_country_flag = FRA_CGT_ban_sorelians }
			}
			add_to_variable = {
				var = FRA_CGT_SOR_strength_var
				value = FRA_size_var
				tooltip = FRA_CGT_SOR_tt
			}
		}
	}
	else = {
		add_to_variable = {
			var = FRA_CGT_ANARSYN_strength_var
			value = FRA_size_var
			tooltip = FRA_CGT_ANARSYN_tt
		}			
		if = {
			limit = {
				NOT = { has_country_flag = FRA_CGT_ban_sorelians }
			}
			add_to_variable = {
				var = FRA_CGT_SOR_strength_var
				value = FRA_size_var
				tooltip = FRA_CGT_SOR_tt
			}
		}
		add_to_variable = { FRA_CGT_SYN_strength_var_dummy = FRA_size_var }
		add_to_variable = { FRA_CGT_ANAR_strength_var_dummy = FRA_size_var }
	}
	FRA_update_percentage_effect = yes
}

FRA_phalanstere_development_in_random_state = {
	random_owned_state = {
		limit = {
			impassable = no
			is_in_home_area = yes
			free_building_slots = {
				building = industrial_complex
				size > 0
				include_locked = yes
			}
			NOT = {
				has_state_category = zero
				has_state_category = wasteland
				has_state_category = zero_island
			}
		}
		add_extra_state_shared_building_slots = 1
		add_building_construction = {
			type = industrial_complex
			level = 1
			instant_build = yes
		}
	}
}
