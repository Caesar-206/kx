NFL_decrease_depression = {
	subtract_from_variable = {
		var = NFL_depression
		value = 1
	}
	add_to_variable = {
		var = NFL_depression_recovery
		value = 1
	}
	custom_effect_tooltip = NFL_decrease_depression
	if = {
		limit = { has_idea = NFL_great_depression }
		swap_ideas = {
			remove_idea = NFL_great_depression
			add_idea = NFL_great_depression_2
		}
	}
	else_if = {
		limit = { has_idea = NFL_great_depression_2 }
